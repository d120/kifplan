\documentclass{article}
\usepackage{namensschilder}

\newcommand{\schilder}[4]{
  \Namensschild{#1}{#2}{#3}{#4}
  \Namensschild{#1}{#2}{#3}{#4}
}


\begin{document}


{% for kiffel in kiffels %}

  {% if kiffel.ist_anonym %}

    \schilder{ ~ }{ ~ }{  ~  }{ {{kiffel.get_barcode}} }

  {% elif kiffel.ist_orga %}

    \schilder{ {{kiffel.nickname|default:'~'}} }{ {{kiffel.hochschule|default:'~'}} }{ ORGA }{ {{kiffel.get_barcode}} }

  {% else %}

    {% if kiffel.ist_kiffel %}
      \schilder{ {{kiffel.nickname|default:'~'}} }{ {{kiffel.hochschule|default:'~'}} }{ ~ }{ {{kiffel.get_barcode}} }
    {% endif %}

    {% if kiffel.ist_helfer %}
      \schilder{ {{kiffel.nickname|default:'~'}} }{ {{kiffel.hochschule|default:'~'}} }{ HELFER }{ {{kiffel.get_barcode}} }
    {% endif %}

  {% endif %}

{% endfor %}

\end{document}
